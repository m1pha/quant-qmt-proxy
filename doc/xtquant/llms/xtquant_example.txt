## 1 行情示例

### 1.1 获取行情示例

要点：
- 先用 `download_history_data`（增量）补充历史K线数据
- 下载财务数据、板块数据
- 用 `get_market_data_ex` 获取本地历史+实时行情
- 通过 `subscribe_quote` 订阅实时行情，使用 `xtdata.run()` 阻塞

示例代码：

```python
from xtquant import xtdata
import time

code_list = ['000001.SZ']
period = '1d'

# 1. 下载数据
for i in code_list:
    xtdata.download_history_data(i, period=period, incrementally=True)

# 2. 财务、板块补充
xtdata.download_financial_data(code_list)
xtdata.download_sector_data()

# 3. 读取历史行情
history_data = xtdata.get_market_data_ex([], code_list, period=period, count=-1)
print(history_data)
print('=' * 20)

# 4. 订阅实时行情
for i in code_list:
    xtdata.subscribe_quote(i, period=period, count=-1)

# 5. 等待订阅
time.sleep(1)

kline_data = xtdata.get_market_data_ex([], code_list, period=period)
print(kline_data)

for i in range(10):
    kline_data = xtdata.get_market_data_ex([], code_list, period=period)
    print(kline_data)
    time.sleep(3)

# 6. 回调模式

def f(data):
    # data 是一次触发的更新数据（通常1条）
    kline_in_callback = xtdata.get_market_data_ex([], code_list, period=period)
    print(kline_in_callback)

for i in code_list:
    xtdata.subscribe_quote(i, period=period, count=-1, callback=f)

xtdata.run()
```

### 1.2 连接VIP服务器示例

```python
import time
import pandas as pd
from xtquant import xtdatacenter as xtdc
from xtquant import xtdata

xtdc.set_token('这里输入token')

addr_list = [
    '115.231.218.73:55310',
    '115.231.218.79:55310',
    '42.228.16.211:55300',
    '42.228.16.210:55300',
    '36.99.48.20:55300',
    '36.99.48.21:55300',
]
xtdc.set_allow_optmize_address(addr_list)

xtdc.set_kline_mirror_enabled(True)
xtdc.init()
port = xtdc.listen(port=58621)

```

### 1.3 连接指定服务器示例

```python
import time
from xtquant import xtdata

info = {'ip': '115.231.218.73', 'port': 55300, 'username': '', 'pwd': ''}

connect_success = 0

def func(d):
    global connect_success
    if d.get('ip') == info['ip'] and d.get('port') == info['port']:
        connect_success = 1 if d.get('status') == 'connected' else 2

xtdata.watch_quote_server_status(func)
qs = xtdata.QuoteServer(info)
qs.connect()

while connect_success == 0:
    time.sleep(0.3)

if connect_success == 2:
    print('连接失败')

print('当前连接状态', xtdata.get_quote_server_status())
```

### 1.4 下载＋全推订阅示例

```python
import time
from xtquant import xtdata

code='600000.SH'
print('全推数据', xtdata.get_full_tick([code]))
xtdata.download_history_data(code, period='1m', start_time='20230701')

def callback_func(data):
    print('回调触发', data)

xtdata.subscribe_quote(code, period='1m', count=-1, callback=callback_func)

data = xtdata.get_market_data(['close'], [code], period='1m', start_time='20230701')
print('一次性取数据', data)
xtdata.run()
```

### 2.4 交易接口重连示例

示例核心：

- 主循环里保持 `xt_trader` 连接
- 断线重连时注意 `session_id` 不要无限增长

### 2.5 指定session id范围连接

```python
def connect(path, session):
    from xtquant import xttrader
    trader = xttrader.XtQuantTrader(path, session)
    trader.start()
    return trader if trader.connect() == 0 else None

ids=list(range(100,200))
import random
random.shuffle(ids)

for session_id in ids:
    trader = connect('userdata_mini', session_id)
    if trader:
        print('连接成功', session_id)
        break
```
